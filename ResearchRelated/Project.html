def abbrev_synthspec(obj,spec_file):
    '''
    obj - corresponds to the object of interest in the SDSS DR16Q catalog
    '''
    
    path1 = './Data/'
    tfile_path = "./"

    DR16Q = fits.open(path1 + "spAll-v5_13_0.fits", mmap = True)
    hdr = DR16Q[1].header
    data = DR16Q[1].data

    theta   = data[obj]["THETA"]
    npoly   = data[obj]["NPOLY"]
    tcolumn = data[obj]["TCOLUMN"]
    tfile   = data[obj]["TFILE"]
    z       = data[obj]["Z"]
    
    spec = fits.open(path1 + spec_file)
    
    loglam  = spec[1].data['loglam']        
    flux = spec[1].data['flux']
    original_spec = [loglam,flux]
    
    starflux = fits.open(tfile_path + tfile, mmap = True)
    shdr  = starflux[0].header
    sdata = starflux[0].data

    starloglam0 = shdr["COEFF0"]
    stardloglam = shdr["COEFF1"]

    ndim = sdata.ndim
    dims = sdata.shape
    npixstar = dims[1]

    ncol = 0
    for i in tcolumn:
        if i != -1:
            ncol += 1

    sf = sdata[tcolumn[:ncol],]

    sf = np.column_stack((sf.transpose(),poly_array(npixstar,npoly)))

    synflux = sf @ theta[0:ncol + npoly]

    starloglam = starloglam0 + np.arange(npixstar) * stardloglam

    starloglam += np.log10(1 + z)
    
    return synflux, starloglam, original_spec

def poly_array(npix,npoly):
    
    arr = np.zeros([npix,npoly])
    xvec = np.arange(npix) / npix
    for i in range(npoly):
        arr[:,i]= xvec**i
    
    return arr
